```typescript
              <button

                </h3>
                {expandedId === faq.faqNumber.toString() ? (
                  <ChevronUp className="w-5 h-5 text-gray-600 flex-shrink-0 ml-4" />
                ) : (
                  <ChevronDown className="w-5 h-5 text-gray-600 flex-shrink-0 ml-4" />
                )}
              </button>

              {expandedId === faq.faqNumber.toString() && (
                <div className="px-6 py-4 bg-gray-50 border-t border-gray-200">
                  <p className="text-gray-700 whitespace-pre-wrap">
                    {faq.answer}
                  </p>
                </div>
              )}
            </div>
          ))}
        </div>
      </div>
    </section>
  )
}
```

## 3. CREATE API ENDPOINT (src/app/api/faqs/route.ts)

Add new API endpoint to fetch FAQs from Sanity:

```typescript
import { client } from '@/sanity/lib/client'
import { NextRequest, NextResponse } from 'next/server'

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url)
    const tier = searchParams.get('tier')
    const pageLocation = searchParams.get('pageLocation')

    let query = '*[_type == "faq" && published == true]'

    if (tier) {
      query += ` | filter(tier == "${tier}")`
    }

    if (pageLocation) {
      query += ` | filter("${pageLocation}" in displayPages)`
    }

    query += ' | sort(priority asc)'

    const faqs = await client.fetch(query)

    return NextResponse.json(faqs)
  } catch (error) {
    console.error('FAQ API error:', error)
    return NextResponse.json({ error: 'Failed to fetch FAQs' }, { status: 500 })
  }
}
```
